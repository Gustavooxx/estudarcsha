using System;
using System.Globalization;

namespace estudarcsha;

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("PROGRAMA DO AÇAÍ COM CUPOM");

        string[] tamanhos = { "pequenos", "médios", "grandes" };
        int[] quantidades = new int[tamanhos.Length];

        for (int i = 0; i < tamanhos.Length; i++)
        {
            Console.WriteLine($"Informe quantos {tamanhos[i]} você irá querer:");
            quantidades[i] = Convert.ToInt32(Console.ReadLine());
        }

        Console.WriteLine("Informe a porcentagem do cupom (ex.: 10 para 10%):");
        double cupom = Convert.ToDouble(Console.ReadLine());

        double total = Resu(quantidades[0], quantidades[1], quantidades[2], cupom);

        Console.WriteLine($"O total foi de {total.ToString("C", CultureInfo.GetCultureInfo("pt-BR"))}");
    }

    public static double Resu(int n1, int n2, int n3, double cupom)
    {
        int[] quant = { n1, n2, n3 };
        double[] precos = { 13.5, 15.0, 17.5 };

        double soma = 0;
        for (int i = 0; i < quant.Length; i++)
        {
            soma += quant[i] * precos[i];
        }

        double resu = soma * (1 - (cupom / 100.0));
        if (resu < 0) resu = 0; // Evita total negativo caso cupom > 100
        return resu;
    }
}

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("PROGRAMA DO SALARIO");

        Console.WriteLine("Informe seu salario base");
        double salario = Convert.ToDouble(Console.ReadLine());

        Console.WriteLine("Informe seu bonus");
        int bonus = Convert.ToInt32(Console.ReadLine());

        Console.WriteLine("informe seu desconto");
        double desc = Convert.ToDouble(Console.ReadLine());

        double resu = Resu(salario, bonus, desc);

        Console.WriteLine($"Seu salario liquido e de {resu}");
    }

    public static double Resu(double salario, int bonus, double desc)
    {
        double soma = ((bonus * 100) + salario) - desc;
        return soma;
    }
}

    {
        int totalSegundos = paginas * paginasPorSegundo;

using System;

namespace estudarcsha;

class Program
{
    static void Main(string[] args)
    {
        Apresentacao();
        double diametro1 = DiametroCilindro1();
        double altura1 = AlturaCilindro1();
        double diametro2 = DiametroCilindro2();
        double altura2 = AlturaCilindro2();
        VerificarTransferencia(diametro1, altura1, diametro2, altura2);
    }

    public static void Apresentacao()
    {
        Console.WriteLine("PROGRAMA DE VERIFICAÇÃO DE TRANSFERÊNCIA DE VOLUME ENTRE CILINDROS");
    }

    public static double DiametroCilindro1()
    {
        Console.WriteLine("Informe o diâmetro do primeiro cilindro:");
        double diametro = Convert.ToDouble(Console.ReadLine());
        return diametro;
    }

    public static double AlturaCilindro1()
    {
        Console.WriteLine("Informe a altura do primeiro cilindro:");
        double altura = Convert.ToDouble(Console.ReadLine());
        return altura;
    }

    public static double DiametroCilindro2()
    {
        Console.WriteLine("Informe o diâmetro do segundo cilindro:");
        double diametro = Convert.ToDouble(Console.ReadLine());
        return diametro;
    }

    public static double AlturaCilindro2()
    {
        Console.WriteLine("Informe a altura do segundo cilindro:");
        double altura = Convert.ToDouble(Console.ReadLine());
        return altura;
    }

    public static double CalcularVolume(double diametro, double altura)
    {
        double raio = diametro / 2.0;
        return Math.PI * raio * raio * altura;
    }

    public static void VerificarTransferencia(double diametro1, double altura1, double diametro2, double altura2)
    {
        double volume1 = CalcularVolume(diametro1, altura1);
        double volume2 = CalcularVolume(diametro2, altura2);
        Console.WriteLine($"O primeiro cilindro tem volume de {volume1}");
        Console.WriteLine($"O segundo cilindro tem volume de {volume2}");
        bool possivel = volume1 <= volume2;
        Console.WriteLine($"É possível transferir o primeiro para o segundo? {possivel.ToString().ToLower()}");
    }
}
